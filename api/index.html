Course Enrollment

Request

$.ajax({
  url: '/enroll',
  method: 'POST',
  data: {
    courseId: 'COURSE_ID',
    userId: 'USER_ID',
  },
  success: function(response) {
    console.log(response);
  },
  error: function(error) {
    console.error(error);
  }
});
